Svarfil for automatisering av innstallasjon
Syntaxen blir UNIK_NAVN = SVAR - Kommentar,utfyllende opplysninger
Dersom du skal ha lengre strenger så må dette settes i ""

# openldap.sh
DC = (ldap suffix)
IP = (intern ip)

# gateway.sh
SQUIPORT = (portnummer på squid innstallasjon)

# bind9.sh
DOMAIN = (domenenavn)
IPREV = (reversert ip addresse, subnetta)


# dhcp3.sh
DOMAIN = (domenet til intern nettet)
DNSOTHER = (ip adresse på dns tjener)
SUBNET = (subnettet til intern nett)
NETMASK = (netmaska til intern nettet)
RANGE_START = (første ip adresse i dhcp range)
RANGE_STOP = (siste ip adresse i dhcp range)
ROUTER_IP = (ip adresse til ruter på intern nett)

# squid.sh
INTERNIP= (ip adresse på internt nett)
PORT= (portnummer på squid tjeneren)
NETTMASKE= (nettmaske for logging)
CIDR= (intern nett på CIDR form)

# apache2.sh
NETT_I_CIDR= (Nettadressen med CIDR notasjon)
MYSQLROOTPASS= (Trenger denne for at wordpress databasen skal opprettes.)

# samba.sh
LDAPSUFFIX= (Full bane i ldapsyntaks. dc=example,dc=no)
WORKGROUP (Domenet som windowsklienter skal melde seg inn i. Må ikke blandes med domenenavnet til tjeneren.)
NETBIOSNAME (Maskinnavnet til tjeneren satt i smb.conf)
LDAPPASSWORD (Et passord)

# email.sh
HOSTNAVN= (Navnet som blir satt under installasjon eller navnet man har endret til etter installasjon )
DOMENE= (tihlde.org feks)   
HOSTIP= (Ekstern IP)   
EKSTERNT_INTERFACE= (eth0 feks)
